57void SensorState::UpdateFret(uint8_t ks0, uint8_t ks1, uint8_t ks2)
58{
59 // Efficiently check FretBoard status registers for highest pressed fret
60 _fret = 0;
61
62 if (ks2)
63 {
64 if (ks2 & 0x40) _fret = 19;
65 else if (ks2 & 0x20) _fret = 18;
66 else if (ks2 & 0x10) _fret = 17;
67 else if (ks2 & 0x08) _fret = 16;
68 else if (ks2 & 0x04) _fret = 15;
69 else if (ks2 & 0x02) _fret = 14;
70 else if (ks2 & 0x01) _fret = 13;
71 }
72 else if (ks1)
73 {
74 if (ks1 & 0x08) _fret = 12;
75 else if (ks1 & 0x04) _fret = 11;
76 else if (ks1 & 0x02) _fret = 10;
77 else if (ks1 & 0x01) _fret = 9;
78 }
79 else if (ks0)
80 {
81 if (ks0 & 0x80) _fret = 8;
82 else if (ks0 & 0x40) _fret = 7;
83 else if (ks0 & 0x20) _fret = 6;
84 else if (ks0 & 0x10) _fret = 5;
85 else if (ks0 & 0x08) _fret = 4;
86 else if (ks0 & 0x04) _fret = 3;
87 else if (ks0 & 0x02) _fret = 2;
88 else if (ks0 & 0x01) _fret = 1;
89 else _fret = 0;
90 }
91 if (_fret != _prevFret)
92 {
93 _isScreenUpdate = true;
94 _prevFret = _fret;
95 }
96}

175bool SensorState::UpdateRotEnc(uint8_t newValue)
176{
177 uint8_t constrainedRotEnc;
178
179 /* Sample Rotary Encoder Twist Knob */
180 if (_isLefty)
181 {
182 _rotEnc = _prevRotEnc + (-1 * (newValue - _prevRotEnc));
183 }
184 else
185 {
186 _rotEnc = newValue;
187 }
188
189 constrainedRotEnc = constrain(_rotEnc, ROT_ENC_MIN, ROT_ENC_MAX);
190 if (constrainedRotEnc != _prevRotEnc)
191 {
192 _isScreenUpdate = true;
193 _prevRotEnc = constrainedRotEnc;
194 return true;
195 }
196 return false;
197}
