100void loop()
101{
102 if (fretBoard.isValueUpdate())
103 {
104 keyStatus0 = fretBoard.QT2120ReadSingleReg(REG_QT2120_KEY_STATUS_0);
105 keyStatus1 = fretBoard.QT2120ReadSingleReg(REG_QT2120_KEY_STATUS_1);
106 keyStatus2 = fretBoard.QT1070ReadSingleReg(REG_QT1070_KEY_STATUS_0);
107 state.UpdateFret(keyStatus0, keyStatus1, keyStatus2);
108 }
109
110 if (strumBoard.isValueUpdate())
111 {
112 strumStatus0 = strumBoard.QT2120ReadSingleReg(REG_QT2120_KEY_STATUS_0);
113 strumStatus1 = strumBoard.QT2120ReadSingleReg(REG_QT2120_KEY_STATUS_1);
114 strumStatus2 = strumBoard.QT1070ReadSingleReg(REG_QT1070_KEY_STATUS_0);
115 state.UpdateStrumKey(strumStatus0, strumStatus1, strumStatus2);
116 }
117
118 state.UpdateRotPot();
119 state.UpdateRotEncSwitch();
120
121 // handle rotary encoder state
122 if (state.UpdateRotEnc(RotaryEncoder.read()))
123 {
124 RotaryEncoder.write(state.GetRotEncValue());
125 }
126
127 // Get Ultrasonic Distance sensor reading
128 if (micros() >= ping_time)
129 {
130 // due to using newPing timer, this has to indirectly set range_in_us
131 Ultrasonic.ping_timer(pingCheck);
132 range_in_cm = range_in_us / US_ROUNDTRIP_CM; // NOTE this US_ROUNDTRIP_CM is in NewPing source code
133 ping_time += ULTRASONIC_PING_PERIOD_MICROS;
134 }
135
136 // constrain range_in_cm, but sufficiently low values are treated as high ones
137 if (range_in_cm < PITCH_BEND_MIN_CM || range_in_cm > PITCH_BEND_MAX_CM)
138 {
139 range_in_cm = PITCH_BEND_MAX_CM;
140 }
141
142 state.UpdateUltrasonic(ONEBYTE_SCALED_PITCH_BEND(range_in_cm));
143
144 // Check Lefty Flip status
145 accel.Update();
146 state.SetIsLeftyFlipped(accel.IsLeftyFlipped());
147 state.UpdateXYZ(accel.x, accel.y, accel.z);
148
149 // if any variables changed this iter, wipe and update screen
150 state.CheckUpdateScreen();
151 delay(100); //
152}
