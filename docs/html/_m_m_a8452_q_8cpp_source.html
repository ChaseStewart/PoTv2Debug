62int8_t MMA8452Q::init(void)
63{
64 byte c = readRegister(MMA8452Q_WHOAMI_REG); // Read WHOAMI register
65 if (c == MMA8452Q_WHOAMI_VAL) // WHOAMI should always be 0x2A
66 {
67 Serial.println("INFO: MMA8452Q is online...");
68 }
69 else
70 {
71 Serial.print("WARNING: Could not connect to MMA8452Q: 0x");
72 Serial.println(c, HEX);
73 Serial.println("WARNING: Failed to init accelerometer!");
74 return 0;
75 }
76
77 MMA8452Standby(); // Must be in standby to change registers
78
79 byte fsr = GSCALE; // Set up the full scale range to 2, 4, or 8g.
80 if(fsr > 8) fsr = 8; //Easy error check
81 fsr >>= 2; // Neat trick, see page 22. 00 = 2G, 01 = 4A, 10 = 8G
82 writeRegister(MMA8452Q_XYZ_DATA_CFG_REG, fsr);
83
84 // The default data rate is 800Hz and we don't modify it in this example code
85
86 MMA8452Active(); // Set to active to start reading
87 return 1;
88}

145static byte readRegister(byte addressToRead)
146{
147 Wire1.beginTransmission(MMA8452Q_SLAVE_ADDR);
148 Wire1.write(addressToRead);
149 Wire1.endTransmission(false); //endTransmission but keep the connection active
150
151 Wire1.requestFrom(MMA8452Q_SLAVE_ADDR, 1); //Ask for 1 byte, once done, bus is released by default
152
153 while(!Wire1.available()) ; //Wait for the data to come back
154 return Wire1.read(); //Return this one byte
155}

168static void readRegisters(byte addressToRead, int bytesToRead, byte * dest)
169{
170 Wire1.beginTransmission(MMA8452Q_SLAVE_ADDR);
171 Wire1.write(addressToRead);
172 Wire1.endTransmission(false); //endTransmission but keep the connection active
173
174 Wire1.requestFrom(MMA8452Q_SLAVE_ADDR, bytesToRead); //Ask for bytes, once done, bus is released by default
175
176 while(Wire1.available() < bytesToRead); //Hang out until we get the # of bytes we expect
177
178 for(int x = 0 ; x < bytesToRead ; x++)
179 {
180 dest[x] = Wire1.read();
181 }
182}
