66void QTouchBoard::_InitQT1070()
67{
68 // Get Chip ID
69 uint8_t chipId = _ReadSingleReg(false, REG_QT1070_CHIP_ID);
70 Serial.print("QT1070 chipId = 0x"); Serial.print(chipId, HEX); Serial.print(", should be 0x"); Serial.println(VAL_QT1070_CHIP_ID, HEX);
71
72 // Get FW version
73 uint8_t versionByte = _ReadSingleReg(false, REG_QT1070_VERSION);
74 uint8_t versionMajor = (versionByte & 0xF0) >> 4;
75 uint8_t versionMinor = (versionByte & 0x0F);
76 Serial.print("Firmware version = "); Serial.print(versionMajor); Serial.print("."); Serial.println(versionMinor);
77
78 // Set touch integration
79 for (int i=0; i<7; i++)

81 _WriteSingleReg(false, i + REG_QT1070_INTEGRATION, 4);
82 _WriteSingleReg(false, i + REG_QT1070_AVE_AKS, 0x20);
83 }
84

130uint8_t QTouchBoard::_ReadSingleReg(bool isQTouch2120, uint8_t reg)
131{
132 uint8_t i2cAddr = (isQTouch2120) ? QTOUCH2120_ADDR : QTOUCH1070_ADDR;
133
134 _i2cStream->beginTransmission(i2cAddr);
135 _i2cStream->write(reg);
136 _i2cStream->endTransmission(false); // endTransmission but keep the connection active
137
138 _i2cStream->requestFrom((int) i2cAddr, 1); // Ask for 1 byte, once done, bus is released by default
139
140 while(!_i2cStream->available()) ; // Wait for the data to come back
141 return _i2cStream->read(); // Return this one byte
142}
